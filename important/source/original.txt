local CustomUILib = {}

function CustomUILib:MakeWindow(Config)
    local title = Config.Name or "My Custom UI"
    
    -- 安全なUIの出力先を決定
    local player = game:GetService("Players").LocalPlayer
    local success, coreGui = pcall(function() return game:GetService("CoreGui") end)
    local targetGui = (success and coreGui) and coreGui or player:WaitForChild("PlayerGui")
    
    -- 古いUIが残っていたら削除
    if targetGui:FindFirstChild("MyExternalUI") then
        targetGui.MyExternalUI:Destroy()
    end
    
    -- メイン画面の作成
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "MyExternalUI"
    ScreenGui.Parent = targetGui
    
    local MainFrame = Instance.new("Frame", ScreenGui)
    MainFrame.Size = UDim2.new(0, 350, 0, 200)
    MainFrame.Position = UDim2.new(0.5, -175, 0.5, -100)
    MainFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    MainFrame.Active = true
    MainFrame.Draggable = true -- ドラッグ移動可能
    Instance.new("UICorner", MainFrame).CornerRadius = UDim.new(0, 6)
    
    -- タイトルの作成
    local Title = Instance.new("TextLabel", MainFrame)
    Title.Size = UDim2.new(1, 0, 0, 35)
    Title.BackgroundTransparency = 1
    Title.Text = "   " .. title
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.TextXAlignment = Enum.TextXAlignment.Left
    Title.Font = Enum.Font.GothamBold
    Title.TextSize = 14
    
    -- 要素を入れるコンテナ
    local Container = Instance.new("Frame", MainFrame)
    Container.Size = UDim2.new(1, -20, 1, -45)
    Container.Position = UDim2.new(0, 10, 0, 35)
    Container.BackgroundTransparency = 1
    
    local ListLayout = Instance.new("UIListLayout", Container)
    ListLayout.Padding = UDim.new(0, 8)
    
    local Window = {}
    
    -- スライダー（数値入力式）を作る関数
    function Window:AddSlider(SliderConfig)
        local name = SliderConfig.Name or "Slider"
        local min = SliderConfig.Min or 0
        local max = SliderConfig.Max or 100
        local def = SliderConfig.Default or 50
        local callback = SliderConfig.Callback or function() end
        
        local SliderFrame = Instance.new("Frame", Container)
        SliderFrame.Size = UDim2.new(1, 0, 0, 35)
        SliderFrame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        Instance.new("UICorner", SliderFrame).CornerRadius = UDim.new(0, 4)
        
        local Label = Instance.new("TextLabel", SliderFrame)
        Label.Size = UDim2.new(0.6, 0, 1, 0)
        Label.Position = UDim2.new(0, 10, 0, 0)
        Label.BackgroundTransparency = 1
        Label.Text = name
        Label.TextColor3 = Color3.fromRGB(220, 220, 220)
        Label.TextXAlignment = Enum.TextXAlignment.Left
        Label.Font = Enum.Font.GothamSemibold
        Label.TextSize = 13
        
        local TextBox = Instance.new("TextBox", SliderFrame)
        TextBox.Size = UDim2.new(0.3, 0, 0.7, 0)
        TextBox.Position = UDim2.new(0.65, 0, 0.15, 0)
        TextBox.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
        TextBox.TextColor3 = Color3.fromRGB(255, 255, 255)
        TextBox.Text = tostring(def)
        TextBox.Font = Enum.Font.Gotham
        Instance.new("UICorner", TextBox).CornerRadius = UDim.new(0, 4)
        
        -- エンターキーで値を反映する処理
        TextBox.FocusLost:Connect(function()
            local val = tonumber(TextBox.Text)
            if val then
                val = math.clamp(val, min, max)
                TextBox.Text = tostring(val)
                callback(val)
            else
                TextBox.Text = tostring(def)
            end
        end)
    end
    
    return Window
end

return CustomUILib
